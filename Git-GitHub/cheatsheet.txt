|------------------------------------------------|
|CREAR UN NUEVO REPOSITORIO                      |
|------------------------------------------------|
|git init                                        |
|------------------------------------------------|


|-------------------------------------------------|
|AÃ‘ADIR UN ARCHIVO                                |
|-------------------------------------------------|
|git add nombre_directorio -> (archivo especifico)|
|-------------------------------------------------|
|git add . -> (todos los archivos)                |
|-------------------------------------------------|


|-------------------------------------------------|
|VERIFICAR EL ESTADO                              |
|-------------------------------------------------|
|git status                                       |
|-------------------------------------------------|


|-------------------------------------------------|
|CONFIGURAR EL USUARIO                            |
|-------------------------------------------------|
|git config user.name "nombre_usuario"            |
|-------------------------------------------------|


|-------------------------------------------------|
|CONFIGURAR EL EMAIL                              |
|-------------------------------------------------|
|git config user.email "nombre@servidor.com       |
|-------------------------------------------------|


|-------------------------------------------------|
|VER LA CONFIGURACION DE GIT                      |
|-------------------------------------------------|
|git config -l                                    |
|-------------------------------------------------|


|-------------------------------------------------|
|OBTENER AYUDA CON EL USO DE COMANDOS             |
|-------------------------------------------------|
|git help                                         |
|-------------------------------------------------|


|-------------------------------------------------|
|VINCULARSE CON EL REPOSITORIO REMOTO             |
|-------------------------------------------------|
|git remote add origin "Url repositorio remoto"   |
|-------------------------------------------------|


|-------------------------------------------------|
|ACTUALIZAR EL REPOSITORIO LOCAL SEGUN            |
|EL REOSITORIO REMOTO                             |
|-------------------------------------------------|
|git pull origin (nombre de la rama)              |
|-------------------------------------------------|


|-------------------------------------------------|
|CLONAR REPOSITORIO REMOTO                        |
|-------------------------------------------------|
|git clone "URL repositorio remoto"               |
|-------------------------------------------------|


|-------------------------------------------------|
|SUBIR CAMBIOS                                    |
|-------------------------------------------------|
|git push origin (nombre de la rama)              |
|-------------------------------------------------|


|-------------------------------------------------|
|POSICIONARSE EN LA RAMA                          |
|-------------------------------------------------|
|git checkout (nombre de la rama)                 |
|-------------------------------------------------|


|-------------------------------------------------|
|VER EL HISTORIAL                                 |
|-------------------------------------------------|
|git log (muestra el historia)                    |
|-------------------------------------------------|
|git log --oneline (muestra el estado -archivos)  |
|-------------------------------------------------|


|-------------------------------------------------|
|VERIFICAR EL REPOSITORIO VINCULADO               |
|-------------------------------------------------|
|git remote -v                                    |
|-------------------------------------------------|


|-------------------------------------------------|
|CREAR RAMA                                       |
|-------------------------------------------------|
|git branch nombre_rama (crea una rama)           |
|-------------------------------------------------|
|git branch -b nombre_rama  (crea una rama        |
|y apunta a ella)                                 |
|-------------------------------------------------|


|-------------------------------------------------|
|LISTAR RAMAS                                     |
|-------------------------------------------------|
|git branch                                       |
|-------------------------------------------------|


|-------------------------------------------------|
|UNIR RAMAS                                       |
|-------------------------------------------------|
|git merge (nueva_rama)                           |
|-------------------------------------------------|


|-------------------------------------------------|
|SUBIR RAMAS AL REPOSITORIO REMOTO                |
|-------------------------------------------------|
|git push --all                                   |
|-------------------------------------------------|


|-------------------------------------------------|
|ELIMINAR EL ULTIMO COMMIT                        |
|-------------------------------------------------|
|git reset --hard HEAD^                           |
|-------------------------------------------------|
|git push origin -f                               |
